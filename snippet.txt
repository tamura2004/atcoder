"# ModInt\n",
"record ModInt, value : Int64 = 0_i64 do\n",
"  MOD = 1_000_000_007_i64\n",
"  def +(other); ModInt.new((@value + other.to_i64 % MOD) % MOD); end\n",
"  def <<(other); ModInt.new((@value << other) % MOD); end\n",
"  delegate inspect, to: @value\n",
"  delegate to_s, to: @value\n",
"  delegate to_i64, to: @value\n",
"end",
